  const overrideMap = require('./api/_assets/trec-20-18-pdflib-overridemap');
  const fixture = require('./scripts/.trec-20-18-fixture.json');
  const { PDFDocument } = require('pdf-lib');
  
  const pdfBytes = fs.readFileSync('api/_assets/trec-20-18-raw.pdf');
  const pdfDoc = await PDFDocument.load(pdfBytes);
  const form = pdfDoc.getForm();
  
  for (const field of form.getFields()) {
    const fixtureKey = overrideMap[field.getName()];
    const value = fixture[fixtureKey];
    if (value) field.setText(value);
  }
  
  const filled = await pdfDoc.save();
  fs.writeFileSync('filled.pdf', filled);
